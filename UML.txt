package observer {

    interface Observer<T> {
        +onUpdate(value : T)
    }

    interface Subject<T> {
        +subscribe(observer : Observer<T>)
        +unsubscribe(observer : Observer<T>)
        +notifyObservers(value : T)
    }

    abstract class AbstractSubject<T>

    Subject <|.. AbstractSubject
    AbstractSubject "1" o-- "*" Observer : notifies
}

package sensor {

    interface RobotSensors {
        +sonar : Sensor<Double>
        +vision : Sensor<Color>
        +temperature : Sensor<Double>
        +lineLeft : Sensor<Boolean>
        +lineCenter : Sensor<Boolean>
        +lineRight : Sensor<Boolean>
        +collision : Sensor<Boolean>
    }

    abstract class Sensor<T> {
        +label : String
        +mountForward : Double
        +mountLateral : Double
        +mountAngle : Double
        +reading : T

        +worldPose(robotPose : Pose) : Pose
        +update(env : Environment, robotPose : Pose)
        #measure(env : Environment, sensorPose : Pose) : T
    }

    class SonarSensor
    class VisionSensor
    class TemperatureSensor
    class LineSensor
    class CollisionSensor

    Sensor --|> AbstractSubject

    Sensor ..> Environment : measures from
    Sensor ..> Pose : uses

    Sensor <|-- SonarSensor
    Sensor <|-- VisionSensor
    Sensor <|-- TemperatureSensor
    Sensor <|-- LineSensor
    Sensor <|-- CollisionSensor
}

package command {

    interface Command {
        +execute()
        +undo()
    }

    interface RobotActuator {
        +leftTrackVelocity : Double
        +rightTrackVelocity : Double
        +setTrackVelocities(left,right)
    }

    class CommandInvoker {
        -undoStack : ArrayDeque<Command>
        -redoStack : ArrayDeque<Command>

        +run(command)
        +undo()
        +redo()

        +canUndo() : Boolean
        +canRedo() : Boolean
    }

    class SetTrackVelocityCommand {
        -receiver : RobotActuator
        -left : Double
        -right : Double
        -previousLeft : Double
        -previousRight : Double

        +execute()
        +undo()
    }

    class RobotCommandFactory {
        +stop(actuator) : Command
        +forward(actuator,speed) : Command
        +reverse(actuator,speed) : Command
        +turnLeft(actuator,speed) : Command
        +turnRight(actuator,speed) : Command
        +setTracks(actuator,left,right) : Command
    }

    CommandInvoker --> Command : executes

    Command ..> RobotActuator : receiver

    Command <|.. SetTrackVelocityCommand
    SetTrackVelocityCommand --> RobotActuator

    RobotCommandFactory ..> SetTrackVelocityCommand : creates
    RobotCommandFactory ..> RobotActuator
}

package api {

    interface RobotApi {
        +sensors : RobotSensors
        +actuator : RobotActuator

        +stop()
        +forward(speed)
        +reverse(speed)
        +turnLeft(speed)
        +turnRight(speed)
        +setTracks(left,right)

        +perform(command)
        +undo()
        +redo()
    }

    class DefaultRobotApi {
        -factory : RobotCommandFactory
    }

    interface ProgramRegistry {
        +register(program)
        +programs()
    }

    class DefaultProgramRegistry

    class StudentPrograms {
        +registerAll(registry)
    }

    RobotApi <|.. DefaultRobotApi

    DefaultRobotApi --> CommandInvoker
    DefaultRobotApi --> RobotActuator
    DefaultRobotApi --> RobotSensors
    DefaultRobotApi --> RobotCommandFactory

    ProgramRegistry --> program.RobotProgram

    DefaultProgramRegistry ..|> ProgramRegistry

    StudentPrograms ..> ProgramRegistry
    StudentPrograms ..> program.RobotProgram
}

package program {

    interface RobotProgram {
        +name : String

        +startProgram(robot : api.RobotApi)
        +stopProgram(robot : api.RobotApi)
    }

    class LineFollowerProgram {

        -robot : api.RobotApi

        -leftObserver : observer.Observer<Boolean>
        -centerObserver : observer.Observer<Boolean>
        -rightObserver : observer.Observer<Boolean>

        +name : String

        +startProgram(robot : api.RobotApi)
        +stopProgram(robot : api.RobotApi)
    }

    class TemperatureSeekProgram {

        -robot : api.RobotApi

        -temperatureObserver : observer.Observer<Double>
        -collisionObserver : observer.Observer<Boolean>

        +name : String

        +startProgram(robot : api.RobotApi)
        +stopProgram(robot : api.RobotApi)
    }

    class BallSeekProgram {

        -robot : api.RobotApi

        -visionObserver : observer.Observer<Color>
        -sonarObserver : observer.Observer<Double>
        -collisionObserver : observer.Observer<Boolean>

        +name : String

        +startProgram(robot : api.RobotApi)
        +stopProgram(robot : api.RobotApi)
    }

    RobotProgram <|.. LineFollowerProgram
    RobotProgram <|.. TemperatureSeekProgram
    RobotProgram <|.. BallSeekProgram

    LineFollowerProgram ..> api.RobotApi : issues actions
    TemperatureSeekProgram ..> api.RobotApi : issues actions
    BallSeekProgram ..> api.RobotApi : issues actions

    LineFollowerProgram ..> observer.Observer : subscribes
    TemperatureSeekProgram ..> observer.Observer : subscribes
    BallSeekProgram ..> observer.Observer : subscribes
}

package model {

    class Robot {

        +pose : Pose

        +leftTrackVelocity : Double
        +rightTrackVelocity : Double

        +setTrackVelocities(left,right)

        +subscribeLeftVelocity(observer)
        +unsubscribeLeftVelocity(observer)

        +subscribeRightVelocity(observer)
        +unsubscribeRightVelocity(observer)

        +step(dt, env)
        +updateSensors(env)
        +stop()
    }

    Robot ..|> RobotActuator
    Robot ..|> RobotSensors

    Robot *-- AbstractSubject : left velocity
    Robot *-- AbstractSubject : right velocity

    Robot *-- SonarSensor
    Robot *-- VisionSensor
    Robot *-- TemperatureSensor

    Robot *-- LineSensor : left
    Robot *-- LineSensor : center
    Robot *-- LineSensor : right

    Robot *-- CollisionSensor
}

}
